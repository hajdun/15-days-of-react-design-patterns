⏱ Debounce Pattern
SearchBox.jsx

API call fires only after user pauses typing for 300ms. Without debounce, every keystroke triggers a search re-computation.

 useDebounce.js
function useDebounce(value, delay = 300) {
  const [debounced, setDebounced] = useState(value);
  useEffect(() => {
    const id = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(id); // cancel on next keystroke
  }, [value, delay]);
  return debounced;
}

⚡ Throttle Pattern
ScrollTracker.jsx

Scroll position updates at most once every 150ms — not on every pixel. Prevents flooding the render cycle on fast scrolls.

 useThrottle.js
function useThrottle(value, limit = 150) {
  const [throttled, setThrottled] = useState(value);
  const lastRun = useRef(Date.now());
  useEffect(() => {
    if (Date.now() - lastRun.current >= limit) {
      lastRun.current = Date.now();
      setThrottled(value);
    }
  }, [value, limit]);
  return throttled;
}

🧠 useMemo + Derived State
No useState for filtered

filtered is derived directly from users + debouncedQuery via useMemo. No useState, no useEffect sync.

 Dashboard.jsx
const users = useMemo(() => generateUsers(10_000), []);

const debouncedQuery = useDebounce(query, 300);

// ✅ Derived — no useState, always in sync
const filtered = useMemo(() =>
  debouncedQuery
    ? users.filter(u =>
        u.name.toLowerCase().includes(debouncedQuery.toLowerCase()) ||
        u.email.toLowerCase().includes(debouncedQuery.toLowerCase()))
    : users,
  [users, debouncedQuery]
);
